# Testing Wishart and Gamma distributions

gamma.wishart <- "
model{
prec.w  ~ dwish(scale.w,df)
prec.g ~ dgamma(scale.g,rate)
}"

#### initial conditions
n_trait = 5
n = 1
scale.w = diag(n,n_traits)
df = n_traits
mean =n*df
scale.g = mean/2
rate = 1/2

### analysis of model and data
update = list(scale.w=scale.w, df=df,scale.g=scale.g,rate=rate)
mod <- jags.model(file=textConnection(gamma.wishart),
                  data=update,
                  n.adapt=1000,n.chains=3)
jdat <- coda.samples(mod,variable.names=c("prec.w","prec.g"),n.iter=10000)
summary(jdat)
